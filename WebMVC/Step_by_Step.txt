    RequestPath = new PathString("/statics")
});

B5: Tạo và viết Controller SinhVien để đưa data danh sách ra bên ngoài
private readonly SinhVienActions _svActions;

public SinhVienController()
{
    _svActions = new SinhVienActions();
}

public IActionResult GetAll()
{
    List<SinhVien> ds = _svActions.GetAll();
    return View(ds);
}
